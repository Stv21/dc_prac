1. IPC using Message Passing

from concurrent.futures import ThreadPoolExecutor
from random import randint
from queue import Queue

q = Queue()

def publisher():
    while True:
        x = "".join([chr(ord('a') + randint(0, 25)) for _ in range(randint(0, 20))])
        print(f"Writing:{x}")
        q.put(x)

def subscriber():
    while True:
        print(f"Reading:{q.get()}")
        q.task_done()

with ThreadPoolExecutor(2) as pool:
    x = pool.submit(publisher)
    pool.submit(subscriber)
    while x.running():
        continue

3. Bully Algorithm

from random import random

class Node:
    def __init__(self, id):
        self.id = id
        self.coord = False

    def get_id(self):
        return self.id

    def becom_coord(self):
        self.coord = True

    def init_elect(self, nodes):
        valid = []
        print(f"Node:{self.id} initing elec")
        for node in nodes:
            rcv = node.receive_elec()
            if rcv and self.get_id() < node.get_id():
                print(f"Node:{self.id} received res from {node.get_id()}")
                valid.append(node)
        if not valid:
            print(f"Node:{self.id} became coord")
            self.becom_coord()
        else:
            valid.sort(key=lambda x: x.id)
            print(f"Node:{self.id} delegating to {valid[-1].id}")
            valid[-1].init_elect(nodes)

    def receive_elec(self):
        ck = random()
        if ck > 0.5:
            return True
        return False

sys = [Node(i) for i in range(10)]
sys[0].init_elect(sys)

4. Ring Algorithm

processes = [1,2,3,4,5]

def ring(start):
	index = processes.index(start)
	n = len(processes)
	
	print(f"process{start} starts election")
	message=[start]
	current = index
	
	while True:
		next_index=(current+1)%n
		next_process = processes[next_index]
		print(f"{processes[current]} -> {next_process} sends {message}")
		
		if next_process not in message:
			message.append(next_process)

		current = next_index
		if processes[current] == start:
            		break
	leader = max(message)
	print(leader)

5. Ricart-Agarwala Algorithm

import threading
import time
import queue

class Node(threading.Thread):
    def __init__(self, node_id, total_nodes):
        super().__init__()
        self.node_id = node_id
        self.total_nodes = total_nodes
        self.clock = 0
        self.request_queue = []  # List of (timestamp, node_id)
        self.acks_received = 0
        self.lock = threading.Lock()
        self.messages = queue.Queue()
        self.running = True

    def log(self, msg):
        print(f"[Node {self.node_id} | Clock {self.clock}]: {msg}")

    def send_message(self, target_node, msg_type, timestamp):
        network[target_node].messages.put((msg_type, timestamp, self.node_id))

    def request_access(self):
        with self.lock:
            self.clock += 1
            request_ts = self.clock
            self.request_queue.append((request_ts, self.node_id))
            self.request_queue.sort()
            self.acks_received = 0
        self.log("Requesting Critical Section...")
        for i in range(self.total_nodes):
            if i != self.node_id:
                self.send_message(i, "REQUEST", request_ts)
        # Wait until top of queue and all ACKs received
        while True:
            with self.lock:
                is_first = self.request_queue[0][1] == self.node_id
                if is_first and self.acks_received == self.total_nodes - 1:
                    break
            time.sleep(0.1)
        self.enter_critical_section()

    def enter_critical_section(self):
        self.log("!!! ENTERED CRITICAL SECTION !!!")
        time.sleep(1)  # Simulate work
        self.log("Leaving Critical Section.")
        with self.lock:
            self.request_queue.pop(0)
            release_ts = self.clock
        for i in range(self.total_nodes):
            if i != self.node_id:
                self.send_message(i, "RELEASE", release_ts)

    def run(self):
        while self.running:
            try:
                msg_type, ts, sender_id = self.messages.get(timeout=1)
                with self.lock:
                    # Lamport Clock update: C = max(C, incoming_ts) + 1
                    self.clock = max(self.clock, ts) + 1
                    if msg_type == "REQUEST":
                        self.request_queue.append((ts, sender_id))
                        self.request_queue.sort()
                        self.send_message(sender_id, "ACK", self.clock)
                    elif msg_type == "ACK":
                        self.acks_received += 1
                    elif msg_type == "RELEASE":
                        self.request_queue = [r for r in self.request_queue if r[1] != sender_id]
            except queue.Empty:
                continue

# Setup simulation
num_nodes = 3
network = [Node(i, num_nodes) for i in range(num_nodes)]
for n in network:
    n.start()

# Node 0 and Node 2 both want access at roughly the same time
time.sleep(1)
threading.Thread(target=network[0].request_access).start()
threading.Thread(target=network[2].request_access).start()

# Let it run then shut down
time.sleep(5)
for n in network:
    n.running = False

8. Round Robin Load Balancing

class RoundRobinLoadBalancer:
    def __init__(self, servers):
        self.servers = servers
        self.index = 0

    def get_server(self):
        server = self.servers[self.index]
        self.index = (self.index + 1) % len(self.servers)
        return server

servers = ["Server A", "Server B", "Server C"]
lb = RoundRobinLoadBalancer(servers)

print("Round Robin Load Balancing:")
for i in range(10):
    print(f"Request {i+1} handled by {lb.get_server()}")
