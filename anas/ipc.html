from concurrent.futures import ThreadPoolExecutor
from random import randint
from queue import Queue

q = Queue()

def publisher():
    while True:
        x = "".join([chr(ord('a') + randint(0, 25)) for _ in range(randint(0, 20))])
        print(f"Writing:{x}")
        q.put(x)

def subscriber():
    while True:
        print(f"Reading:{q.get()}")
        q.task_done()

with ThreadPoolExecutor(2) as pool:
    x = pool.submit(publisher)
    pool.submit(subscriber)
    while x.running():
        continue
