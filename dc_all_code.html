3. Bully Algorithm

processes = [ 1,2,3,4,5]
alive = [1,2,3,4]

def bulli(ini):
	print(f"the ini is {ini} process")
	candidates=[p for p in processes if p>ini and p in alive]

	if not candidates:
		print(f"Process{ini} is coordinator")
		return ini
	else:
		return bulli(max(candidates))


neta= bulli(2)
print("neta:",neta)

4. Ring Algorithm

processes = [1,2,3,4,5]

def ring(start):
	index = processes.index(start)
	n = len(processes)
	
	print(f"process{start} starts election")
	message=[start]
	current = index
	
	while True:
		next_index=(current+1)%n
		next_process = processes[next_index]
		print(f"{processes[current]} -> {next_process} sends {message}")
		
		if next_process not in message:
			message.append(next_process)

		current = next_index
		if processes[current] == start:
            		break
	leader = max(message)
	print(leader)
ring(3)

5. Ricart-Agarwala Algorithm

processes = [1, 2, 3]
requesting = []

def request_cs(pid):
    requesting.append(pid)
    print(f"Process {pid} requesting Critical Section")

def grant_cs(pid):
    others = [p for p in requesting if p != pid]
    if not others or pid < min(others):
        print(f"Process {pid} granted Critical Section")
        requesting.remove(pid)
    else:
        print(f"Process {pid} deferred by Process {min(others)}")

for p in processes:
    request_cs(p)

for p in processes:
    grant_cs(p)

request = {'P3': 'R1'}

def detect_deadlock():
    for p, r in request.items():

7. Random Load Balancing

import random

servers = ["S1", "S2", "S3"]

def random_lb():
    for i in range(5):
        server = random.choice(servers)
        print(f"Request {i} -> {server}")

random_lb()

8. Round Robin Load Balancing

servers = ["S1", "S2", "S3"]

def round_robin(n):
    for i in range(n):
        server = servers[i % len(servers)]
        print(f"Request {i} -> {server}")

round_robin(6)

9. Least Connection Load Balancing

servers = { "S1":0,"S2":0,"S3":0}
def least(n):
	for i in range(n):
		server=min(servers,key=servers.get)
		servers[server]+=1
		print(f"request{i} assigned to {server}")
least(6)
