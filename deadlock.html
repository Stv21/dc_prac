processes = ['P1', 'P2', 'P3']
resources = ['R1', 'R2']

allocation = {'P1': 'R1', 'P2': 'R2'}
request = {'P3': 'R1'}

def detect_deadlock():
    for p, r in request.items():
        for ap, ar in allocation.items():
            if r == ar:
                print(f"Deadlock possible: {p} waiting for {ap}")

detect_deadlock()
